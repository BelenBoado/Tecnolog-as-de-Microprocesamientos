.include "m328pdef.inc"

.org 0x0000
rjmp inicio

.equ f_cpu = 16000000
.equ baud = 9600
.equ bps = (F_CPU/16/baud) - 1

inicio:
       ldi r16, high(ramend)
       out SPH, r16
       ldi r16, low(ramend)
       out SPL, r16

       ldi r16, low(bps)
       ldi r17, high(bps)
       rcall initUART

menu:
     call preguntar
     call getc
     call delay  

wait:
     call getc
     cpi r16, 0
     breq wait
     ldi r18, 1
     add r16, r18
     rcall putc
     ldi r16, 0
     rjmp wait

initUART:
         sts UBRR0L, r16
         sts UBRR0H, r17
         ldi r16, (1<<RXEN0) | (1<<TXEN0)
         sts UCSR0B, r16
         ret

putc:
     lds r17, UCSR0A
     sbrs r17, UDRE0
     rjmp putc
     sts UDR0, r16
     ldi r16, 0
     ret

getc:
     lds r17, UCSR0A
     sbrs r17, UDRE0
     rjmp getc
     lds r16, UDR0
     ret
