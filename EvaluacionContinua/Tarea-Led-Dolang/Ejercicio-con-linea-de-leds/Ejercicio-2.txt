.include "m328pdef.inc"
.org 0x00

; Inicializaci√≥n
ldi r16, 0xfe
ldi r20 ,0xff
out DDRD,r20

main_loop:
    out PORTD, r16
    rcall delay

    lsl r16
    cpi r16, 0x00
    brne main_loop

    out PORTD, r16
    rjmp main_loop


delay:
    ldi r18, 70
    ldi r19, 254
    ldi r21, 254
L1:
    dec r21
    brne L1
    dec r19
    brne L1
    dec r18
    brne L1
    ret
